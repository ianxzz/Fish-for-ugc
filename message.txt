local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local SellRemote = ReplicatedStorage:WaitForChild("SellFishRequest")

local player = Players.LocalPlayer
local backpack = player:WaitForChild("Backpack")
local running = false

-- Função que vende o peixe
local function sellFish(fishName, amount)
    local args = {
        {
            name = fishName,
            action = "doOne",
            amount = amount or 1
        }
    }
    
    SellRemote:FireServer(unpack(args))
end

-- Vende todos os peixes atualmente na backpack
local function processBackpack()
    local fishCount = {}

    for _, tool in ipairs(backpack:GetChildren()) do
        if tool:IsA("Tool") and tool.Name ~= "FishRod" then
            local itemId = tool:GetAttribute("ItemId")
            if itemId then
                fishCount[itemId] = (fishCount[itemId] or 0) + 1
            end
        end
    end

    for fishName, amount in pairs(fishCount) do
        sellFish(fishName, amount)
    end
end

-- Detector de peixes entrando na Backpack
local function setupInstantSell()
    backpack.ChildAdded:Connect(function(child)
        if not running then return end
        
        if child:IsA("Tool") and child.Name ~= "FishRod" then
            local itemId = child:GetAttribute("ItemId")
            if itemId then
                -- vende 1 imediatamente
                sellFish(itemId, 1)
            end
        end
    end)
end

local function mainLoop()
    task.spawn(function()
        while running do
            processBackpack()
            task.wait(0.3)
        end
    end)
end

Window:Toggle({
    Text = "Auto Sell",
    Callback = function(v)
        running = v
        print("Pablovitar:", v)

        if running then
            setupInstantSell()
            mainLoop()
        end
    end
})
